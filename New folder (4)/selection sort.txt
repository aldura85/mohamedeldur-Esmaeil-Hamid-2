#include <stdio.h>
#include<time.h>                 
#include<stdlib.h>                
int main()
{
    srand(time(NULL));                       
   int array[100000], n = 1000, c, d, position, swap;

   

   printf("Enter %d integers\n", n);

   for ( c = 1 ; c < n ; c++ )
      array[c]=rand()%1000 + 1;               

   for ( c = 1 ; c < ( n - 1 ) ; c++ )
   {
      position = c;

      for ( d = c + 1 ; d < n ; d++ )
      {
         if ( array[position] > array[d] )
            position = d;
      }
      if ( position != c )
      {
         swap = array[c];
         array[c] = array[position];
         array[position] = swap;
      }
   }

   printf("Sorted list in ascending order:\n");

   for ( c = 1 ; c < n ; c++ )
      printf("%d\n", array[c]);

   return 0;
}
